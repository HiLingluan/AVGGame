ViewModelMgr = class("ViewModelMgr")

function ViewModelMgr:ctor()
    self:initProperty()
end

function  ViewModelMgr:initProperty()
   self.viewModelList = {}  
end

function  ViewModelMgr:register(key,viewModel)
    if not self.viewModelList[key] then
        self.viewModelList[key] = viewModel
    end
end

function  ViewModelMgr:remove(key)
    if self.viewModelList[key] then
        self.viewModelList[key] = nil
    end
end

--根据key获取viewModel
function ViewModelMgr:GetViewModel(key)
    if self.viewModelList[key] then
        return self.viewModelList[key]
    end
    LogWarn("找不到ViewModel==="..tostring(key))
    return nil
end

ViewModelMgr:ctor()
